[[1,2]] -> jsonToDel();##
[
  "1,2"
]

$arrays1 = [[1,2],[3,4]];##
"$arrays1"

$arrays1 -> jsonToDel();##
[
  "1,2",
  "3,4"
]

$arrays1 ->jsonToDel()->write(stdout());##
{
  "type": "stdout"
}

$arrays2 = [['one','two'],['three','four']];##
"$arrays2"

$arrays2 -> jsonToDel();##
[
  "\"one\",\"two\"",
  "\"three\",\"four\""
]

$arrays2 -> jsonToDel() -> write(stdout());##
{
  "type": "stdout"
}

$records = [{name: 'mike', age: 10}, {name: 'john', age: 20}];##
"$records"

$records -> jsonToDel({fields: ['name', 'age']});##
[
  "\"mike\",10",
  "\"john\",20"
]

$records -> jsonToDel({fields: ['name', 'age']}) -> write(stdout());##
{
  "type": "stdout"
}

$mixed =  [{name: 'mike', age: 10}, [100, 200]];##
"$mixed"

$mixed -> jsonToDel({fields: ['name', 'age'], delimiter: '<=>'});##
[
  "\"mike\"<=>10",
  "100<=>200"
]

$mixed -> jsonToDel({fields: ['name', 'age'], delimiter: '<=>'}) -> write(stdout());##
{
  "type": "stdout"
}
